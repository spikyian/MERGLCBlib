120// EVENT DECODING
121// An event opcode has bits 4 and 7 set, bits 1 and 2 clear
122// An ON event opcode also has bit 0 clear
123// An OFF event opcode also has bit 0 set
124//
125// eg:
126// ACON 90 1001 0000
127// ACOF 91 1001 0001
128// ASON 98 1001 1000
129// ASOF 99 1001 1001
130//
131// ACON1 B0 1011 0000
132// ACOF1 B1 1011 0001
133// ASON1 B8 1011 1000
134// ASOF1 B9 1011 1001
135//
136// ACON2 D0 1101 0000
137// ACOF2 D1 1101 0001
138// ASON2 D8 1101 1000
139// ASOF2 D9 1101 1001
140//
141// ACON3 F0 1111 0000
142// ACOF3 F1 1111 0001
143// ASON3 F8 1111 1000
144// ASOF3 F9 1111 1001
145//
146// ON/OFF determined by d0
147// Long/Short determined by d3
148// num data bytes determined by d7,d6
149//
150#define EVENT_SET_MASK 0b10010000
151#define EVENT_CLR_MASK 0b00000110
152#define EVENT_ON_MASK 0b00000001
153#define EVENT_SHORT_MASK 0b00001000
